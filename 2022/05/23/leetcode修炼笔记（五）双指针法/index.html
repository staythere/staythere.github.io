leetcode修炼笔记（五）双指针法

PYF · 2022-5-23 · 次阅读

题五：双指针法

leading:代码随想录 https://programmercarl.com/

知识点

双指针法：一般是指的在遍历对象的过程中，不是使用单个指针进行访问，而是使用两个相同方向或者相反方向的指针进行扫描，从而达到相应的目的。

一般双指针法有两种表现方式：

  1. 同向移动：在同向移动时，指针互相之间间隔一个距离进行移动
  2. 相向移动：在相向移动中，双指针一个指针在开头，另外一个指针在结尾，根据满足的条件进行移动指针；

题目：

一、移除元素27. 移除元素 - 力扣（LeetCode）

思路：设置快慢指针从前向后遍历，当遇到val时慢指针不动，快指针向前移动，将快指针元素移动到慢指针位置。返回为慢指针位置。

class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int number = nums.size();
        int n = number;  //记录输出长度
        int slowindex = 0;
        int fastindex = 0;

        while(fastindex != number){
            if(nums[fastindex] == val){
                fastindex++;
                continue;
            }else{
                nums[slowindex] = nums[fastindex];
                slowindex++;
                fastindex++;
            }
        }
        return slowindex;
    }
};

二、反转字符串344. 反转字符串 - 力扣（LeetCode）

思路：设置快慢指针，慢指针=0，快指针=s.size(),循环当慢<快时，交换元素。

class Solution {
public:
    void reverseString(vector<char>& s) {
        int slowindex = 0;
        int fastindex = s.size()-1;
        char temp;

        while(slowindex < fastindex){
            temp = s[slowindex];
            s[slowindex] = s[fastindex];
            s[fastindex] = temp;
            slowindex++;
            fastindex--; 
        }
    }
};

三、替换空格剑指 Offer 05. 替换空格 - 力扣（LeetCode）

思路：首先扩充数组到每个空格替换成”%20”之后的大小。然后从后向前替换空格，也就是双指针法

class Solution {
public:
    string replaceSpace(string s) {
        int old_size = s.size();
        int count = 0;
        for(int i = 0; i < s.size(); i++){
            if(s[i] == ' ')count++;
        }
        s.resize(old_size + 2*count);

        int slowindex = old_size-1;
        int fastindex = s.size()-1;
        while(slowindex != fastindex){
            if(s[slowindex] != ' '){
                s[fastindex] = s[slowindex];
                fastindex--;
                slowindex--;
                continue;
            }else{
                s[fastindex--] = '0';
                s[fastindex--] = '2';
                s[fastindex--] = '%';
                slowindex--;
                continue;
            }
        }
        return s;
    }
};

四、颠倒字符串中的单词151. 颠倒字符串中的单词 - 力扣（LeetCode）

思路：

  • 移除多余空格
  • 将整个字符串反转
  • 将每个单词反转
class Solution {
public:
    string reverseWords(string s) {
        int old_size = s.size();
        int new_size = removeElseSpace(s,old_size);

        int slowindex = 0;
        int fastindex = 0;

        //反转整个串
        reverse(s,0,s.size());
        //反转各个单词
        while(fastindex < s.size()){
            if(s[fastindex] == ' '){
                reverse(s,slowindex,fastindex);
                slowindex = fastindex + 1;
            }
            fastindex++;
        }
        reverse(s,slowindex,fastindex); //反转最后一个单词
        return s;
    }


    int removeElseSpace(string& s, int old_size)
    {
        int slowindex = 0;
        int fastindex = 0;
        //去除前面的空格
        while(s.size() > 0 && fastindex < s.size() && s[fastindex] == ' '){
            fastindex++;
        }
        //去除中间的空格
        while(fastindex != old_size){
            if(s[fastindex] == ' ' &&
             fastindex - 1 > 0 && s[fastindex - 1] == s[fastindex] ){
                 fastindex++;
            }else{
                s[slowindex++] = s[fastindex++];
            }
        }

        //结尾只可能有一个空格或没有，去掉一个空格
        if(s[slowindex - 1] == ' ' && slowindex - 1 > 0){
            s.resize(slowindex - 1);
            return slowindex - 1;
        }else{
            s.resize(slowindex);
            return slowindex;
        }
    }

    void reverse(string& s, int begin, int end)
    {
        int slowindex = begin;
        int fastindex = end - 1; //end的位置为空格
        char temp;

        while(slowindex < fastindex){
            temp = s[slowindex];
            s[slowindex] = s[fastindex];
            s[fastindex] = temp;
            slowindex++;
            fastindex--;
        }
    }

};
